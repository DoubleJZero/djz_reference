CREATE TABLE TB_CODE_M
(
	CODE_SEQ			INT(10) AUTO_INCREMENT PRIMARY KEY COMMENT '코드 일련번호'
	, CODE_GROUP		VARCHAR(100) NOT NULL COMMENT '코드그룹'
	, CODE_ID			VARCHAR(10) NOT NULL COMMENT '코드아이디'
	, CODE_DATA			VARCHAR(200) NOT NULL COMMENT '코드데이터'
	, CODE_ORDER		VARCHAR(2) NOT NULL COMMENT '코드순서'
	, CODE_ETC1			VARCHAR(200) NOT NULL COMMENT '기타1'
	, CODE_ETC2			VARCHAR(200) NOT NULL COMMENT '기타2'
	, CODE_ETC3			VARCHAR(200) NOT NULL COMMENT '기타3'
	, CREATE_USER		VARCHAR(16) COMMENT '등록자'
  	, CREATE_DATE		DATE COMMENT '등록일'
  	, UPDATE_USER		VARCHAR(16) COMMENT '수정자'
  	, UPDATE_DATE		DATE COMMENT '수정일'
) COMMENT '코드테이블';
