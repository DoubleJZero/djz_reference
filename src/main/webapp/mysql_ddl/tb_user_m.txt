CREATE TABLE TB_USER_M
(
	USER_ID				VARCHAR(16) PRIMARY KEY COMMENT '아이디'
	, USER_PW			VARCHAR(500) NOT NULL COMMENT '비밀번호'
	, USER_NAME			VARCHAR(100) NOT NULL COMMENT '사용자명'
	, ZIPCODE			VARCHAR(6) NOT NULL COMMENT '우편번호'
	, ADDRESS			VARCHAR(300) NOT NULL COMMENT '주소'
	, ADDRESS_DETAIL	VARCHAR(50) NOT NULL COMMENT '상세주소'
	, EMAIL				VARCHAR(50) NOT NULL COMMENT '이메일'
	, MOBILE			VARCHAR(13) NOT NULL COMMENT '휴대폰번호'
	, SMS_YN			VARCHAR(1) DEFAULT 'N' COMMENT 'sms수신여부'
	, EMAIL_YN			VARCHAR(1) DEFAULT 'N' COMMENT 'email수신여부'
	, CHANGE_PW_DATE	VARCHAR(10) COMMENT '비밀번호변경일'
	, LAST_LOGIN_DATE	DATE COMMENT '마지막 접속일시'
	, CREATE_USER		VARCHAR(16) COMMENT '등록자'
  	, CREATE_DATE		DATE COMMENT '등록일'
  	, UPDATE_USER		VARCHAR(16) COMMENT '수정자'
  	, UPDATE_DATE		DATE COMMENT '수정일'
) COMMENT '사용자테이블';
