CREATE TABLE TB_MENU_M
(
	MENU_SEQ				INT(10) AUTO_INCREMENT PRIMARY KEY COMMENT '메뉴 일련번호'
	, MENU_GROUP			VARCHAR(4) NOT NULL COMMENT '메뉴 그룹'
	, PARENT_MENU_GROUP		VARCHAR(4) DEFAULT '' NOT NULL COMMENT '부모 메뉴 그룹'
	, MENU_DEPTH			INT(3) NOT NULL COMMENT '메뉴 depth'
	, MENU_ID				VARCHAR(7) NOT NULL COMMENT '메뉴 아이디'
	, MENU_NAME 			VARCHAR(100) NOT NULL COMMENT '메뉴명'
	, MENU_DESCRIPTION		VARCHAR(500) NOT NULL COMMENT '메뉴설명'
	, USE_YN				CHAR(1) DEFAULT 'N' NOT NULL COMMENT '사용여부'
	, MENU_URL				VARCHAR(200)  COMMENT '메뉴 URL'
	, MENU_ORDER			INT(3) COMMENT '메뉴 순서'
	, CREATE_USER			VARCHAR(16) COMMENT '등록자'
  	, CREATE_DATE			DATE COMMENT '등록일'
  	, UPDATE_USER			VARCHAR(16) COMMENT '수정자'
  	, UPDATE_DATE			DATE COMMENT '수정일'
) COMMENT '메뉴 테이블';
