CREATE TABLE TB_AUTH_MENU
(
	AUTH_ID 				VARCHAR(20) COMMENT '권한 아이디'
	, MENU_ID 				VARCHAR(7) COMMENT '메뉴 아이디'
	, USE_YN				VARCHAR(1) DEFAULT 'N' COMMENT '사용여부'
	, ROLE_C				VARCHAR(1) DEFAULT 'N' COMMENT '등록권한'
  	, ROLE_R				VARCHAR(1) DEFAULT 'N' COMMENT '조회권한'
  	, ROLE_U				VARCHAR(1) DEFAULT 'N' COMMENT '수정권한'
  	, ROLE_D				VARCHAR(1) DEFAULT 'N' COMMENT '삭제권한'
  	, ROLE_E				VARCHAR(1) DEFAULT 'N' COMMENT '엑셀권한'
  	, ROLE_P				VARCHAR(1) DEFAULT 'N' COMMENT '인쇄권한'
  	, PRIMARY KEY(AUTH_ID, MENU_ID)
) COMMENT '권한별 메뉴기능 테이블';
