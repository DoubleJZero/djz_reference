CREATE TABLE TB_ATCHFL
(
	ATCHFL_ID				VARCHAR(50) PRIMARY KEY COMMENT '파일아이디'
	, ATCHFL_GRP_ID			VARCHAR(50) NOT NULL COMMENT '파일그룹아이디'
	, ATCHFL_NM				VARCHAR(100) NOT NULL COMMENT '파일명'
	, ATCHFL_PHSC_NM		VARCHAR(50) NOT NULL COMMENT '저장 파일명'
	, ATCHFL_ROUTE			VARCHAR(300) NOT NULL COMMENT '저장경로'
	, ATCHFL_EXTNSN_NM		VARCHAR(50) COMMENT '확장자명'
	, ATCHFL_SIZE			VARCHAR(50) COMMENT '파일사이즈'
	, CREATE_USER			VARCHAR(16) COMMENT '등록자'
  	, CREATE_DATE			DATE COMMENT '등록일'
  	, UPDATE_USER			VARCHAR(16) COMMENT '수정자'
  	, UPDATE_DATE			DATE COMMENT '수정일'
) COMMENT '파일테이블';
