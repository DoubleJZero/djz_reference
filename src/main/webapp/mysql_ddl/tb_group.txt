CREATE TABLE TB_GROUP
(
	GROUP_ID 				INT(10) AUTO_INCREMENT PRIMARY KEY COMMENT '그룹 아이디'
	, GROUP_NAME 			VARCHAR(50) COMMENT '그룹명'
	, CREATE_USER			VARCHAR(16) COMMENT '등록자'
  	, CREATE_DATE			DATE COMMENT '등록일'
  	, UPDATE_USER			VARCHAR(16) COMMENT '수정자'
  	, UPDATE_DATE			DATE COMMENT '수정일'
) COMMENT '그룹 테이블';
